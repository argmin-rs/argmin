14/// # Wolfe conditions
15///
16/// Assures that a step length satisfies a "sufficient decrease" in cost function value (see
17/// [`ArmijoCondition`](`crate::solver::linesearch::condition::ArmijoCondition`) as well as that
18/// the slope has been reduced sufficiently (thus making it more likely to be close to a local
19/// minimum).
20#[derive(Clone, Copy, Debug, Default, Eq, PartialEq)]
21#[cfg_attr(feature = "serde1", derive(Serialize, Deserialize))]
22pub struct WolfeCondition<F> {
23    c1: F,
24    c2: F,
25}
26
27impl<F> WolfeCondition<F>
28where
29    F: ArgminFloat,
30{
31    /// Construct a new instance of [`WolfeCondition`].
32    ///
33    /// # Example
34    ///
35    /// ```
36    /// # use argmin::solver::linesearch::condition::WolfeCondition;
37    /// let wolfe = WolfeCondition::new(0.0001f64, 0.1f64);
38    /// ```
39    pub fn new(c1: F, c2: F) -> Result<Self, Error> {
40        if c1 <= float!(0.0) || c1 >= float!(1.0) {
41            return Err(argmin_error!(
42                InvalidParameter,
43                "WolfeCondition: Parameter c1 must be in (0, 1)"
44            ));
45        }
46        if c2 <= c1 || c2 >= float!(1.0) {
47            return Err(argmin_error!(
48                InvalidParameter,
49                "WolfeCondition: Parameter c2 must be in (c1, 1)"
50            ));
51        }
52        Ok(WolfeCondition { c1, c2 })
53    }
54}
55
56impl<T, G, F> LineSearchCondition<T, G, F> for WolfeCondition<F>
57where
58    G: ArgminDot<T, F>,
59    F: ArgminFloat,
60{
61    fn evaluate_condition(
62        &self,
63        current_cost: F,
64        current_gradient: Option<&G>,
65        initial_cost: F,
66        initial_gradient: &G,
67        search_direction: &T,
68        step_length: F,
69    ) -> bool {
70        let tmp = initial_gradient.dot(search_direction);
71        (current_cost <= initial_cost + self.c1 * step_length * tmp)
72            && current_gradient
73                .expect(concat!(
74                    "Current gradient not supplied to",
75                    " `evaluate_condition` method of `WolfeCondition`"
76                ))
77                .dot(search_direction)
78                >= self.c2 * tmp
79    }
80
81    fn requires_current_gradient(&self) -> bool {
82        true
83    }
84}

93    #[test]
94    fn test_wolfe_new() {
95        let c1: f64 = 0.01;
96        let c2: f64 = 0.08;
97        let WolfeCondition {
98            c1: c1_wolfe,
99            c2: c2_wolfe,
100        } = WolfeCondition::new(c1, c2).unwrap();
101        assert_eq!(c1.to_ne_bytes(), c1_wolfe.to_ne_bytes());
102        assert_eq!(c2.to_ne_bytes(), c2_wolfe.to_ne_bytes());
103
104        // c1
105        assert_error!(
106            WolfeCondition::new(1.0, 0.5),
107            ArgminError,
108            "Invalid parameter: \"WolfeCondition: Parameter c1 must be in (0, 1)\""
109        );
110
111        assert_error!(
112            WolfeCondition::new(0.0, 0.5),
113            ArgminError,
114            "Invalid parameter: \"WolfeCondition: Parameter c1 must be in (0, 1)\""
115        );
116
117        assert_error!(
118            WolfeCondition::new(-1.0, 0.5),
119            ArgminError,
120            "Invalid parameter: \"WolfeCondition: Parameter c1 must be in (0, 1)\""
121        );
122
123        assert_error!(
124            WolfeCondition::new(2.0, 0.5),
125            ArgminError,
126            "Invalid parameter: \"WolfeCondition: Parameter c1 must be in (0, 1)\""
127        );
128
129        // c2
130        assert_error!(
131            WolfeCondition::new(0.5, -1.0),
132            ArgminError,
133            "Invalid parameter: \"WolfeCondition: Parameter c2 must be in (c1, 1)\""
134        );
135
136        assert_error!(

145            "Invalid parameter: \"WolfeCondition: Parameter c2 must be in (c1, 1)\""

147

161    #[test]
162    fn test_wolfe() {
163        let c1: f64 = 0.5;
164        let c2: f64 = 0.9;
165        let cond = WolfeCondition::new(c1, c2).unwrap();
166        let f = |x: f64, y: f64| x.powf(2.0) + y.powf(2.0);
167        let g = |x: f64, y: f64| vec![2.0 * x, 2.0 * y];
168        let initial_x = -1.0;
169        let initial_y = -0.0;
170        let search_direction = vec![1.0, 0.0];
171        for (alpha, acc) in [
172            (0.001, false),
173            (0.03, false),
174            (0.1 - f64::EPSILON, false),
175            (0.1, true),
176            (0.5, true),
177            (0.9, true),
178            (0.99, true),
179            (1.0, true),
180            (1.0 + f64::EPSILON, false),
181            (1.5, false),
182            (1.8, false),
183            (2.0, false),
184            (2.3, false),
185        ] {
186            assert_eq!(
187                cond.evaluate_condition(
188                    f(initial_x + alpha, initial_y),
189                    Some(&g(initial_x + alpha, initial_y)),
190                    f(initial_x, initial_y),
191                    &g(initial_x, initial_y),
192                    &search_direction,
193                    alpha,
194                ),
195                acc
196            );
197        }
198    }
